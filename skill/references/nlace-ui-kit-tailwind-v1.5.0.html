Block — componente React

Button.tsx TypeScript 
// Botón NLACE — variantes con Tailwind
const Button = ({ variant = 'primary', children }) => {
  const base = `inline-flex items-center rounded-pill
    font-semibold transition-all duration-ui ease-nl
    hover:-translate-y-0.5 hover:shadow-hover
    px-[22px] py-[11px] text-[0.9rem]`

  const variants = {
    primary:  'bg-nl-primary text-white',
    accent:   'bg-nl-accent text-white',
    secondary:'bg-white border border-nl-border-ui',
  }

  return <button className={`${base} ${variants[variant]}`}>
    {children}
  </button>
}
